MySQL 5.6编译常见参数选项说明

  • -DCMAKE_INSTALL_PREFIX=/application/mysql-5.6 # 设定mysql安装目录
  • -DMYSQL_DATADIR=/application/mysql-5.6/data # 设定mysql数据文件目录
  • -DMYSQL_UNIX_ADDR=/application/mysql-5.6/tmp/mysql.sock # 设定mysql.sock路径
  • -DDEFAULT_CHARSET=utf8 # 设定默认字符集为utf8
  • -DDEFAULT_COLLATION=utf8_general_ci # 设定默认排序规则
  • -DEXTRA_CHARSETS=all # 启用额外的字符集类型
  • -DWITH_INNOBASE_STORAGE_ENGINE=1 # 启用本地数据导入支持
  • -DWITH_FEDERATED_STORAGE_ENGINE=1 # 引擎支持
  • -DWITH_BLACKHOLE_STORAGE_ENGINE=1 # 引擎支持
  • -DWITHOUT_EXAMPLE_STORAGE_ENGINE=1 # 引擎支持
  • -DWITH_ZLIB=bundled # 启用libz库支持
  • -DWITH_SSL=bundled # 启动ssl库支持
  • -DENABLED_LOCAL_INFILE=1 # 启用本地数据库导入支持
  • -DWITH_EMBEDDED_SERVER=1 # 编译嵌入式服务器支持
  • -DWITH_DEBUG=0 # 禁用debug(默认为禁用)
